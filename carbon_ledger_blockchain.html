Script de Déploiement — network.sh
#!/bin/bash # network.sh — Déploiement complet du réseau Carbon Ledger Fabric set -e # 1. Générer les artefacts cryptographiques echo "🔐 Génération des certificats MSP..." cryptogen generate --config=./crypto-config.yaml # 2. Créer le bloc genesis de l'orderer echo "📦 Création du bloc genesis..." configtxgen -profile CarbonOrdererGenesis \ -channelID system-channel \ -outputBlock ./channel-artifacts/genesis.block # 3. Créer la transaction de création du channel configtxgen -profile CarbonChannel \ -outputCreateChannelTx ./channel-artifacts/carbonchannel.tx \ -channelID carbonchannel # 4. Lancer tous les conteneurs echo "🐳 Démarrage des services Docker..." docker-compose up -d sleep 5 # 5. Créer le channel echo "🌐 Création du channel carbonchannel..." peer channel create -o orderer.carbon.com:7050 \ -c carbonchannel \ -f ./channel-artifacts/carbonchannel.tx \ --tls --cafile $ORDERER_CA # 6. Joindre les peers au channel peer channel join -b carbonchannel.block CORE_PEER_ADDRESS=peer1.carbonchannel.com:8051 peer channel join -b carbonchannel.block # 7. Déployer le chaincode echo "📜 Packaging et déploiement du chaincode..." peer lifecycle chaincode package carbon-ledger.tar.gz \ --path ./chaincode/carbon-ledger \ --lang golang \ --label carbon-ledger_1.0 peer lifecycle chaincode install carbon-ledger.tar.gz peer lifecycle chaincode approveformyorg ... peer lifecycle chaincode commit ... echo "✅ Réseau Carbon Ledger opérationnel!"
